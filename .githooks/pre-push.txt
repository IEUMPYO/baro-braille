#!/bin/bash
# Pre-push hook: 빌드 검증
# 실패 시 푸시 차단

echo "[Pre-push] 프로덕션 빌드 검증 실행 중..."

# 1. 프로덕션 빌드 테스트
echo "  - Next.js 빌드 실행"
npm run build

if [ $? -ne 0 ]; then
    echo "ERROR: 프로덕션 빌드 실패"
    exit 1
fi

# 2. 빌드 결과물 확인
if [ ! -d "out" ]; then
    echo "ERROR: 빌드 결과물(out/) 생성 실패"
    exit 1
fi

echo "[Pre-push] 빌드 검증 완료"
